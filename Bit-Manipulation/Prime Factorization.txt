class Solution{
	public:
	
	vector<int>AllPrimeFactors(int N) {
	    vector<int> res;
	    int i=2;
	    while(i<=sqrt(N)){
	        if(N%i==0){
	            res.push_back(i);
	        }
	        while(N%i==0){
	            N=N/i;
	        }
	        i++;
	    }
	    if(N!=1){
	        res.push_back(N);
	    }
	    return res;
	}
};

//{ Driver Code Starts.
int main(){
	int tc;
	cin >> tc;
	while(tc--){
		long long int N;
		cin >> N;
		Solution ob;
		vector<int>ans = ob.AllPrimeFactors(N);
		for(auto i: ans)
			cout << i <<" ";
		cout <<"\n";
	}  
	return 0;
}
