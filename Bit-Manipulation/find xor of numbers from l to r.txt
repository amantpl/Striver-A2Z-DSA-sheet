class Solution {
  public:
  int xor1(int n){
      if(n%4==1) return 1;
      else if(n%4==2) return n+1;
      else if(n%4==3) return 0;
      else return n;
  }
    int findXOR(int l, int r) {
        int res1=xor1(l-1);
        int res2=xor1(r);
        return res1 ^ res2;
    }
};

//{ Driver Code Starts.
int main() {
    int t = 1;
    cin >> t;

    while (t--) {
        // Input

        int l, r;
        cin >> l >> r;

        Solution obj;
        cout << obj.findXOR(l, r) << endl;
    }
}
