class Solution{
    public:
    vector<long long int> twoOddNum(long long int Arr[], long long int N)  
    {
        int xorr=0;
        int rightmost=0;
        for(int i=0;i<N;i++){
            xorr=xorr^Arr[i];
            rightmost=(xorr&(xorr-1))^xorr;
        }
        int b1=0;
        int b2=0;
        for(int i=0;i<N;i++){
            if(Arr[i]&rightmost){
                b1=b1^Arr[i];
            }else{
                b2=b2^Arr[i];
            }
        }
        return {max(b1,b2),min(b1,b2)};
    }
};
