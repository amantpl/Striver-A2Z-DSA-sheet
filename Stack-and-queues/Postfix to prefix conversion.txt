class Solution {
  public:
    string postToPre(string post_exp) {
        int n=post_exp.size();
        stack<string> st;
        for(int i=0;i<n;i++){
            if((post_exp[i]>='a'&&post_exp[i]<='z')||(post_exp[i]>='A'&&post_exp[i]<='Z')){
                st.push(string(1,post_exp[i]));
            }else{
                string top=st.top();
                st.pop();
                string prev_top=st.top();
                st.pop();
                string t=post_exp[i]+prev_top+top;
                st.push(t);
            }
        }
        return st.top();
    }
};
